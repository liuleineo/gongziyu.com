fs 核心模块

什么是 fs 模块？

fs 是 Node.js 的核心模块之一，提供了丰富的文件系统操作API。它支持异步和同步两种操作模式，满足不同场景的需求。

// 导入 fs 模块
const fs = require('fs');

// 导入 promise 版本（推荐）
const fs = require('fs').promises;

// 导入 fs/promises 模块（Node.js 14+）
const fs = require('fs/promises');

异步操作（推荐）

  • • 非阻塞式操作，性能更好
  • • 支持回调函数、Promise 和 async/await
  • • 适合大多数生产环境
  • • 不会阻塞事件循环

同步操作

  • • 阻塞式操作，代码简单
  • • 会阻塞事件循环
  • • 适合初始化操作
  • • 文件名以 Sync 结尾

文件基本操作

创建文件

// 异步创建文件（Promise 版本）
async function createFile() {
  try {
    await fs.writeFile('example.txt', 'Hello, Node.js!');
    console.log('文件创建成功');
  } catch (err) {
    console.error('创建文件失败:', err);
  }
}

// 同步创建文件
try {
  fs.writeFileSync('example.txt', 'Hello, Node.js!');
  console.log('文件创建成功');
} catch (err) {
  console.error('创建文件失败:', err);
}

读取文件

// 异步读取文件（async/await）
async function readFile() {
  try {
    const data = await fs.readFile('example.txt', 'utf8');
    console.log('文件内容:', data);
  } catch (err) {
    console.error('读取文件失败:', err);
  }
}

// 流式读取（大文件推荐）
const readStream = fs.createReadStream('large-file.txt', 'utf8');

readStream.on('data', (chunk) => {
  console.log('读取到数据块:', chunk);
});

readStream.on('end', () => {
  console.log('文件读取完成');
});

修改文件

// 覆盖写入文件
await fs.writeFile('example.txt', '新的内容');

// 追加内容到文件
await fs.appendFile('example.txt', '追加的内容');

// 流式写入
const writeStream = fs.createWriteStream('output.txt');

writeStream.write('第一行内容\n');
writeStream.write('第二行内容\n');
writeStream.end('写入完成');

writeStream.on('finish', () => {
  console.log('文件写入完成');
});

删除文件

// 异步删除文件
async function deleteFile() {
  try {
    await fs.unlink('example.txt');
    console.log('文件删除成功');
  } catch (err) {
    console.error('删除文件失败:', err);
  }
}

// 检查文件是否存在
async function fileExists(filename) {
  try {
    await fs.access(filename, fs.constants.F_OK);
    return true;
  } catch {
    return false;
  }
}

文件夹操作

创建文件夹

// 创建单个文件夹
await fs.mkdir('new-folder');

// 递归创建多级文件夹
await fs.mkdir('path/to/folder', { recursive: true });

读取文件夹

// 读取文件夹内容
const files = await fs.readdir('./');
console.log('文件夹内容:', files);

// 读取文件夹详细信息
const files = await fs.readdir('./', { withFileTypes: true });
files.forEach(file => {
  console.log(file.name, file.isDirectory() ? '文件夹' : '文件');
});

重命名/移动

// 重命名文件或文件夹
await fs.rename('old-name.txt', 'new-name.txt');

// 移动文件
await fs.rename('file.txt', 'new/path/file.txt');

删除文件夹

// 删除空文件夹
await fs.rmdir('empty-folder');

// 递归删除非空文件夹
await fs.rm('non-empty-folder', { recursive: true });

路径处理

path 模块详解

path 模块提供了处理文件路径的工具函数，帮助你在不同操作系统上正确处理文件路径。

const path = require('path');

// 获取文件名
const filename = path.basename('/path/to/file.txt');
// 输出: file.txt

// 获取文件扩展名
const extname = path.extname('/path/to/file.txt');
// 输出: .txt

// 获取目录名
const dirname = path.dirname('/path/to/file.txt');
// 输出: /path/to

// 拼接路径
const fullPath = path.join('/path', 'to', 'file.txt');
// 输出: /path/to/file.txt

// 解析路径
const parsedPath = path.parse('/path/to/file.txt');
// 输出: { root: '/', dir: '/path/to', base: 'file.txt', name: 'file', ext: '.txt' }

常用路径常量

// 当前工作目录
console.log('当前目录:', process.cwd());

// __dirname: 当前模块所在目录的绝对路径
console.log('模块目录:', __dirname);

// __filename: 当前模块文件的绝对路径
console.log('模块文件:', __filename);

// 路径拼接示例
const configPath = path.join(__dirname, 'config', 'app.json');

实用示例

文件复制函数

async function copyFile(source, destination) {
  const readStream = fs.createReadStream(source);
  const writeStream = fs.createWriteStream(destination);

  return new Promise((resolve, reject) => {
    readStream.pipe(writeStream);

    readStream.on('error', reject);
    writeStream.on('error', reject);
    writeStream.on('finish', resolve);
  });
}

// 使用示例
await copyFile('source.txt', 'destination.txt');

遍历目录

async function walkDirectory(dir) {
  const files = await fs.readdir(dir, { withFileTypes: true });
  const result = [];

  for (const file of files) {
    const fullPath = path.join(dir, file.name);
    if (file.isDirectory()) {
      result.push(await walkDirectory(fullPath));
    } else {
      result.push(fullPath);
    }
  }

  return result.flat();
}

// 使用示例
const allFiles = await walkDirectory('./');
console.log('所有文件:', allFiles);

批量重命名

async function batchRename(dir, prefix) {
  const files = await fs.readdir(dir);
  let count = 1;

  for (const file of files) {
    const ext = path.extname(file);
    const newName = `${prefix}-${count}${ext}`;
    await fs.rename(
      path.join(dir, file),
      path.join(dir, newName)
    );
    count++;
  }
}

// 使用示例
await batchRename('./images', 'photo');

错误处理

常见错误类型

ENOENT

文件或目录不存在

EACCES

权限不足

EEXIST

文件已存在

EBUSY

文件正被使用

错误处理最佳实践

async function safeFileOperation() {
  try {
    // 先检查文件是否存在
    try {
      await fs.access('file.txt', fs.constants.F_OK);
      console.log('文件存在');
    } catch (err) {
      console.log('文件不存在，创建新文件');
      await fs.writeFile('file.txt', '默认内容');
    }

    // 执行文件操作
    const data = await fs.readFile('file.txt', 'utf8');
    console.log('文件内容:', data);
  } catch (err) {
    console.error('操作失败:', err.message);
    // 根据错误类型进行不同处理
    if (err.code === 'EACCES') {
      console.error('权限不足，请检查文件权限');
    } else if (err.code === 'ENOENT') {
      console.error('文件或目录不存在');
    }
  }
}

最佳实践

推荐做法

  • • 优先使用异步操作，避免阻塞事件循环
  • • 使用 Promise 和 async/await 处理异步操作
  • • 始终进行错误处理，避免程序崩溃
  • • 使用 path 模块处理路径，确保跨平台兼容性
  • • 对于大文件使用流式操作，减少内存占用
  • • 操作前检查文件/目录是否存在

避免做法

  • • 避免在生产环境使用同步操作
  • • 不要忽略错误处理
  • • 避免手动拼接路径，使用 path.join()
  • • 不要一次性读取大文件到内存
  • • 避免频繁的文件系统操作
  • • 不要假设文件操作总是成功

性能优化

内存优化

使用流式操作处理大文件，避免内存溢出

并发控制

控制并发文件操作数量，避免系统资源耗尽

缓存策略

缓存文件内容，减少重复读取操作
